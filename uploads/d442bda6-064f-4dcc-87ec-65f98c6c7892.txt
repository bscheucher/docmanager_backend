spring init --dependencies=web,data-jpa my-project

# Check available dependencies
spring init --list

# Check available Spring Boot versions
spring init --list | grep "Spring Boot"

# Create project with specific version
spring init \
  --boot-version=3.2.1 \
  --dependencies=web,data-jpa,postgresql,security \
  my-project

# Generate project without downloading (just show what would be created)
spring init --extract my-project

# Create project from start.spring.io with custom settings
spring init \
  --dependencies=web,data-jpa,postgresql \
  --type=maven-project \
  --language=java \
  --packaging=jar \
  --java-version=17 \
  --group-id=com.example \
  --artifact-id=demo \
  --name="Demo Application" \
  --description="Demo project for Spring Boot" \
  --package-name=com.example.demo \
  demo

# If you need to check your Spring CLI version
spring --version

# Get help for the init command
spring help init

# Alternative: Use curl to create project directly from start.spring.io
curl https://start.spring.io/starter.zip \
  -d dependencies=web,data-jpa,postgresql,security,validation,mail,actuator \
  -d type=maven-project \
  -d language=java \
  -d bootVersion=3.2.1 \
  -d groupId=com.deutschinstitut \
  -d artifactId=deutschkurse-api \
  -d name="Deutschkurse Management API" \
  -d description="API for managing German language courses" \
  -d packageName=com.deutschinstitut.deutschkurse \
  -d packaging=jar \
  -d javaVersion=17 \
  -o deutschkurse-api.zip

# Then unzip the downloaded file
unzip deutschkurse-api.zip
